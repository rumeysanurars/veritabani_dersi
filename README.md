CREATE TABLE MUSTERI(
M_NO INT PRIMARY KEY IDENTITY(1,1),
M_ADI NVARCHAR(50),
M_SOYAD NVARCHAR(50),
M_ADRES NVARCHAR(100),
M_TELEFON NVARCHAR(11))

CREATE TABLE ARAC(
ARAC_NO INT IDENTITY(1,1) PRIMARY KEY,
MODEL NVARCHAR (50),
MARKA NVARCHAR(10),
PLAKA NVARCHAR(20),
FIYAT INT)


CREATE TABLE SATIS(
SATIS_NO INT PRIMARY KEY IDENTITY(1,1),
M_NO INT,
ARAC_NO INT,
SATIS_TARIH DATE,
SATIS_FIYAT INT);

CREATE TABLE ALIM(
ALIM_NO INT IDENTITY(1,1) PRIMARY KEY,
M_NO INT,
ARAC_NO INT,
ALIM_TARIH NVARCHAR(20) ,
ALIS_FIYAT INT);
